16  package com.linecorp.armeria.server.annotation;
17  
18  import javax.annotation.Nullable;
19  
20  import com.linecorp.armeria.common.HttpData;
21  import com.linecorp.armeria.common.HttpResponse;
22  import com.linecorp.armeria.common.HttpStatus;
23  import com.linecorp.armeria.common.MediaType;
24  import com.linecorp.armeria.server.ServiceRequestContext;
25  
26  /**
27   * A response converter implementation which creates an {@link HttpResponse} with
28   * {@code content-type: application/binary} or {@code content-type: application/octet-stream}.
29   */
30  public class ByteArrayResponseConverterFunction implements ResponseConverterFunction {
31  
32      @Override
33      public HttpResponse convertResponse(ServiceRequestContext ctx,
34                                          @Nullable Object result) throws Exception {
35          if (result instanceof HttpData) {
36              return HttpResponse.of(HttpStatus.OK, mediaType(ctx.negotiatedResponseMediaType()),
37                                     ((HttpData) result).array());
38          }
39          if (result instanceof byte[]) {
40              return HttpResponse.of(HttpStatus.OK, mediaType(ctx.negotiatedResponseMediaType()),
41                                     (byte[]) result);
42          }
43          return ResponseConverterFunction.fallthrough();
44      }
45  
46      private static MediaType mediaType(@Nullable MediaType mediaType) {
47          if (mediaType == null) {
48              return MediaType.APPLICATION_BINARY;
49          }
50  
51          // A user expects 'binary'.
52          if (mediaType.is(MediaType.APPLICATION_BINARY) ||
53              mediaType.is(MediaType.OCTET_STREAM)) {
54              // @Produces("application/binary") or @ProducesBinary
55              // @Produces("application/octet-stream") or @ProducesOctetStream
56              return mediaType;
57          }
58  
59          return ResponseConverterFunction.fallthrough();
60      }
61  }
