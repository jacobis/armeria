120     /**
121      * Mapping from HTTP method annotation to {@link HttpMethod}, like following.
122      * <ul>
123      *   <li>{@link Options} -> {@link HttpMethod#OPTIONS}
124      *   <li>{@link Get} -> {@link HttpMethod#GET}
125      *   <li>{@link Head} -> {@link HttpMethod#HEAD}
126      *   <li>{@link Post} -> {@link HttpMethod#POST}
127      *   <li>{@link Put} -> {@link HttpMethod#PUT}
128      *   <li>{@link Patch} -> {@link HttpMethod#PATCH}
129      *   <li>{@link Delete} -> {@link HttpMethod#DELETE}
130      *   <li>{@link Trace} -> {@link HttpMethod#TRACE}
131      * </ul>
132      */
133     private static final Map<Class<?>, HttpMethod> HTTP_METHOD_MAP =
134             ImmutableMap.<Class<?>, HttpMethod>builder()
135                     .put(Options.class, HttpMethod.OPTIONS)
136                     .put(Get.class, HttpMethod.GET)
137                     .put(Head.class, HttpMethod.HEAD)
138                     .put(Post.class, HttpMethod.POST)
139                     .put(Put.class, HttpMethod.PUT)
140                     .put(Patch.class, HttpMethod.PATCH)
141                     .put(Delete.class, HttpMethod.DELETE)
142                     .put(Trace.class, HttpMethod.TRACE)
143                     .build();

145     /**
146      * Returns the list of {@link AnnotatedHttpService} defined by {@link Path} and HTTP method annotations
147      * from the specified {@code object}.
148      */
149     static List<AnnotatedHttpServiceElement> find(String pathPrefix, Object object,
150                                                   Iterable<?> exceptionHandlersAndConverters) {
151         Builder<ExceptionHandlerFunction> exceptionHandlers = null;
152         Builder<RequestConverterFunction> requestConverters = null;
153         Builder<ResponseConverterFunction> responseConverters = null;
154 
155         for (final Object o : exceptionHandlersAndConverters) {
156             boolean added = false;
157             if (o instanceof ExceptionHandlerFunction) {
158                 if (exceptionHandlers == null) {
159                     exceptionHandlers = ImmutableList.builder();
160                 }
161                 exceptionHandlers.add((ExceptionHandlerFunction) o);
162                 added = true;
163             }
164             if (o instanceof RequestConverterFunction) {
165                 if (requestConverters == null) {
166                     requestConverters = ImmutableList.builder();
167                 }
168                 requestConverters.add((RequestConverterFunction) o);
169                 added = true;
170             }
171             if (o instanceof ResponseConverterFunction) {
172                 if (responseConverters == null) {
173                     responseConverters = ImmutableList.builder();
174                 }
175                 responseConverters.add((ResponseConverterFunction) o);
176                 added = true;
177             }
178             if (!added) {
179                 throw new IllegalArgumentException(o.getClass().getName() +
180                                                    " is neither an exception handler nor a converter.");
181             }
182         }
183 
184         final List<ExceptionHandlerFunction> exceptionHandlerFunctions =
185                 exceptionHandlers != null ? exceptionHandlers.build() : ImmutableList.of();
186         final List<RequestConverterFunction> requestConverterFunctions =
187                 requestConverters != null ? requestConverters.build() : ImmutableList.of();
188         final List<ResponseConverterFunction> responseConverterFunctions =
189                 responseConverters != null ? responseConverters.build() : ImmutableList.of();
190 
191         final List<Method> methods = requestMappingMethods(object);
192         return methods.stream()
193                       .map((Method method) -> create(pathPrefix, object, method, exceptionHandlerFunctions,
194                                                      requestConverterFunctions, responseConverterFunctions))
195                       .collect(toImmutableList());
196     }

198     /**
199      * Returns an {@link AnnotatedHttpService} instance defined to {@code method} of {@code object} using
200      * {@link Path} annotation.
201      */
202     private static AnnotatedHttpServiceElement create(String pathPrefix, Object object, Method method,
203                                                       List<ExceptionHandlerFunction> baseExceptionHandlers,
204                                                       List<RequestConverterFunction> baseRequestConverters,
205                                                       List<ResponseConverterFunction> baseResponseConverters) {
206 
207         final Set<Annotation> methodAnnotations = httpMethodAnnotations(method);
208         if (methodAnnotations.isEmpty()) {
209             throw new IllegalArgumentException("HTTP Method specification is missing: " + method.getName());
210         }
211 
212         final Set<HttpMethod> methods = toHttpMethods(methodAnnotations);
213         if (methods.isEmpty()) {
214             throw new IllegalArgumentException(method.getDeclaringClass().getName() + '#' + method.getName() +
215                                                " must have an HTTP method annotation.");
216         }
217 
218         final Class<?> clazz = object.getClass();
219         final PathMapping pathMapping = new HttpHeaderPathMapping(
220                 pathStringMapping(pathPrefix, method, methodAnnotations),
221                 methods, consumableMediaTypes(method, clazz), producibleMediaTypes(method, clazz));
222 
223         final List<ExceptionHandlerFunction> eh =
224                 exceptionHandlers(method, clazz).addAll(baseExceptionHandlers)
225                                                 .add(defaultExceptionHandler).build();
226         final List<RequestConverterFunction> req =
227                 requestConverters(method, clazz).addAll(baseRequestConverters).build();
228         final List<ResponseConverterFunction> res =
229                 responseConverters(method, clazz).addAll(baseResponseConverters)
230                                                  .addAll(defaultResponseConverters).build();
231 
232         List<AnnotatedValueResolver> resolvers;
233         try {
234             resolvers = AnnotatedValueResolver.of(method, pathMapping.paramNames(),
235                                                   toRequestObjectResolvers(req));
236         } catch (NoParameterException ignored) {
237             // Allow no parameter like below:
238             //
239             // @Get("/")
240             // public String method1() { ... }
241             //
242             resolvers = ImmutableList.of();
243         }
244 
245         final Set<String> expectedParamNames = pathMapping.paramNames();
246         final Set<String> requiredParamNames =
247                 resolvers.stream()
248                          .filter(AnnotatedValueResolver::isPathVariable)
249                          .map(AnnotatedValueResolver::httpElementName)
250                          .collect(Collectors.toSet());
251 
252         if (!expectedParamNames.containsAll(requiredParamNames)) {
253             final Set<String> missing = Sets.difference(requiredParamNames, expectedParamNames);
254             throw new IllegalArgumentException("cannot find path variables: " + missing);
255         }
256 
257         // Warn unused path variables only if there's no '@RequestObject' annotation.
258         if (resolvers.stream().noneMatch(r -> r.isAnnotationType(RequestObject.class)) &&
259             !requiredParamNames.containsAll(expectedParamNames)) {
260             final Set<String> missing = Sets.difference(expectedParamNames, requiredParamNames);
261             logger.warn("Some path variables of the method '" + method.getName() +
262                         "' of the class '" + clazz.getName() +
263                         "' do not have their corresponding parameters annotated with @Param. " +
264                         "They would not be automatically injected: " + missing);
265         }
266         return new AnnotatedHttpServiceElement(pathMapping,
267                                                new AnnotatedHttpService(object, method, resolvers, eh, res),
268                                                decorator(method, clazz));
269     }

271     /**
272      * Returns the list of {@link Path} annotated methods.
273      */
274     private static List<Method> requestMappingMethods(Object object) {
275         return Arrays.stream(object.getClass().getMethods())
276                      .filter(m -> m.getAnnotation(Path.class) != null ||
277                                   !httpMethodAnnotations(m).isEmpty())
278                      .sorted(Comparator.comparingInt(AnnotatedHttpServiceFactory::order))
279                      .collect(toImmutableList());
280     }

282     /**
283      * Returns the value of the order of the {@link Method}. The order could be retrieved from {@link Order}
284      * annotation. 0 would be returned if there is no specified {@link Order} annotation.
285      */
286     private static int order(Method method) {
287         final Order order = method.getAnnotation(Order.class);
288         return order != null ? order.value() : 0;

291     /**
292      * Returns {@link Set} of HTTP method annotations of a given method.
293      * The annotations are as follows.
294      *
295      * @see Options
296      * @see Get
297      * @see Head
298      * @see Post
299      * @see Put
300      * @see Patch
301      * @see Delete
302      * @see Trace
303      */
304     private static Set<Annotation> httpMethodAnnotations(Method method) {
305         return Arrays.stream(method.getAnnotations())
306                      .filter(annotation -> HTTP_METHOD_MAP.containsKey(annotation.annotationType()))
307                      .collect(Collectors.toSet());
308     }

311      * Returns {@link Set} of {@link HttpMethod}s mapped to HTTP method annotations.
312      *
313      * @see Options
314      * @see Get
315      * @see Head
316      * @see Post
317      * @see Put
318      * @see Patch
319      * @see Delete
320      * @see Trace
321      */
322     private static Set<HttpMethod> toHttpMethods(Set<Annotation> annotations) {
323         return annotations.stream()
324                           .map(annotation -> HTTP_METHOD_MAP.get(annotation.annotationType()))
325                           .filter(Objects::nonNull)
326                           .collect(toImmutableEnumSet());

337     private static List<MediaType> consumableMediaTypes(AnnotatedElement element) {
338         final List<MediaType> mediaTypes = new ArrayList<>();
339 
340         for (final Annotation annotation : element.getAnnotations()) {
341             if (annotation instanceof ConsumesGroup) {
342                 Arrays.stream(((ConsumesGroup) annotation).value())
343                       .forEach(e -> addConsumableMediaType(mediaTypes, MediaType.parse(e.value())));
344             } else if (annotation instanceof Consumes) {
345                 addConsumableMediaType(mediaTypes, MediaType.parse(((Consumes) annotation).value()));
346             } else if (annotation instanceof ConsumeTypes) {
347                 Arrays.stream(((ConsumeTypes) annotation).value())
348                       .forEach(e -> addConsumableMediaType(mediaTypes, MediaType.parse(e.value())));
349             } else if (annotation instanceof ConsumeType) {
350                 addConsumableMediaType(mediaTypes, MediaType.parse(((ConsumeType) annotation).value()));
351             } else {
352                 Arrays.stream(annotation.annotationType().getAnnotationsByType(Consumes.class))
353                       .forEach(e -> addConsumableMediaType(mediaTypes, MediaType.parse(e.value())));
354             }
355         }
356         return mediaTypes;
357     }

359     private static void addConsumableMediaType(List<MediaType> mediaTypes, MediaType newMediaType) {
360         addMediaType(mediaTypes, newMediaType, Consumes.class, true);
361     }

371     private static List<MediaType> producibleMediaTypes(AnnotatedElement element) {
372         final List<MediaType> mediaTypes = new ArrayList<>();
373 
374         for (final Annotation annotation : element.getAnnotations()) {
375             if (annotation instanceof ProducesGroup) {
376                 Arrays.stream(((ProducesGroup) annotation).value())
377                       .forEach(e -> addProducibleMediaType(mediaTypes, MediaType.parse(e.value())));
378             } else if (annotation instanceof Produces) {
379                 addProducibleMediaType(mediaTypes, MediaType.parse(((Produces) annotation).value()));
380             } else if (annotation instanceof ProduceTypes) {
381                 Arrays.stream(((ProduceTypes) annotation).value())
382                       .forEach(e -> addProducibleMediaType(mediaTypes, MediaType.parse(e.value())));
383             } else if (annotation instanceof ProduceType) {
384                 addProducibleMediaType(mediaTypes, MediaType.parse(((ProduceType) annotation).value()));
385             } else {
386                 Arrays.stream(annotation.annotationType().getAnnotationsByType(Produces.class))
387                       .forEach(e -> addProducibleMediaType(mediaTypes, MediaType.parse(e.value())));
388             }
389         }
390         return mediaTypes;
391     }

393     private static void addProducibleMediaType(List<MediaType> mediaTypes, MediaType newMediaType) {
394         addMediaType(mediaTypes, newMediaType, Produces.class, false);
395     }

397     private static void addMediaType(List<MediaType> mediaTypes, MediaType newMediaType,
398                                      Class<?> clazz, boolean allowWildcard) {
399         if (!allowWildcard && newMediaType.hasWildcard()) {
400             throw new IllegalArgumentException('@' + clazz.getSimpleName() + " must not have a wildcard: " +
401                                                newMediaType);
402         }
403         if (mediaTypes.stream().anyMatch(e -> e.equals(newMediaType))) {
404             throw new IllegalArgumentException("Duplicated media type for @" + clazz.getSimpleName() + ": " +
405                                                newMediaType);
406         }
407         mediaTypes.add(newMediaType);
408     }

410     /**
411      * Returns the {@link PathMapping} instance mapped to {@code method}.
412      */
413     private static PathMapping pathStringMapping(String pathPrefix, Method method,
414                                                  Set<Annotation> methodAnnotations) {
415         pathPrefix = ensureAbsolutePath(pathPrefix, "pathPrefix");
416         if (!pathPrefix.endsWith("/")) {
417             pathPrefix += '/';
418         }
419 
420         final String pattern = findPattern(method, methodAnnotations);
421         final PathMapping mapping = PathMapping.of(pattern);
422         if ("/".equals(pathPrefix)) {
423             // pathPrefix is not specified or "/".
424             return mapping;
425         }
426 
427         if (pattern.startsWith(ExactPathMapping.PREFIX)) {
428             return PathMapping.ofExact(concatPaths(
429                     pathPrefix, pattern.substring(ExactPathMapping.PREFIX_LEN)));
430         }
431 
432         if (pattern.startsWith(PrefixPathMapping.PREFIX)) {
433             return PathMapping.ofPrefix(concatPaths(
434                     pathPrefix, pattern.substring(PrefixPathMapping.PREFIX_LEN)));
435         }
436 
437         if (pattern.startsWith(GlobPathMapping.PREFIX)) {
438             final String glob = pattern.substring(GlobPathMapping.PREFIX_LEN);
439             if (glob.startsWith("/")) {
440                 return PathMapping.ofGlob(concatPaths(pathPrefix, glob));
441             } else {
442                 // NB: We cannot use PathMapping.ofGlob(pathPrefix + "/**/" + glob) here
443                 //     because that will extract '/**/' as a path parameter, which a user never specified.
444                 return new PrefixAddingPathMapping(pathPrefix, mapping);
445             }
446         }
447 
448         if (pattern.startsWith(RegexPathMapping.PREFIX)) {
449             return new PrefixAddingPathMapping(pathPrefix, mapping);
450         }
451 
452         if (pattern.startsWith("/")) {
453             // Default pattern
454             return PathMapping.of(concatPaths(pathPrefix, pattern));
455         }
456 
457         // Should never reach here because we validated the path pattern.
458         throw new Error();
459     }

461     /**
462      * Returns a specified path pattern. The path pattern might be specified by {@link Path} or
463      * HTTP method annotations such as {@link Get} and {@link Post}.
464      */
465     private static String findPattern(Method method, Set<Annotation> methodAnnotations) {
466         String pattern = null;
467 
468         final Path path = method.getAnnotation(Path.class);
469         if (path != null) {
470             pattern = method.getAnnotation(Path.class).value();
471         }
472         for (Annotation a : methodAnnotations) {
473             final String p = (String) invokeValueMethod(a);
474             if (DefaultValues.isUnspecified(p)) {
475                 continue;
476             }
477             checkArgument(pattern == null,
478                           "Only one path can be specified. (" + pattern + ", " + p + ')');
479             pattern = p;
480         }
481         if (pattern == null || pattern.isEmpty()) {
482             throw new IllegalArgumentException(
483                     "A path pattern should be specified by @Path or HTTP method annotations.");
484         }
485         return pattern;
486     }

488     /**
489      * Returns a decorator chain which is specified by {@link Decorator} annotations and user-defined
490      * decorator annotations.
491      */
492     private static Function<Service<HttpRequest, HttpResponse>,
493             ? extends Service<HttpRequest, HttpResponse>> decorator(Method method, Class<?> clazz) {
494 
495         final List<DecoratorAndOrder> decorators = collectDecorators(clazz, method);
496 
497         Function<Service<HttpRequest, HttpResponse>,
498                 ? extends Service<HttpRequest, HttpResponse>> decorator = null;
499         for (int i = decorators.size() - 1; i >= 0; i--) {
500             final DecoratorAndOrder d = decorators.get(i);
501             decorator = decorator == null ? d.decorator()
502                                           : decorator.andThen(d.decorator());
503         }
504         return decorator == null ? Function.identity() : decorator;
505     }

507     /**
508      * Returns a decorator list which is specified by {@link Decorator} annotations and user-defined
509      * decorator annotations.
510      */
511     @VisibleForTesting
512     static List<DecoratorAndOrder> collectDecorators(Class<?> clazz, Method method) {
513         final List<DecoratorAndOrder> decorators = new ArrayList<>();
514 
515         // Class-level decorators are applied before method-level decorators.
516         collectDecorators(decorators, clazz.getAnnotations());
517         collectDecorators(decorators, method.getAnnotations());
518 
519         // Sort decorators by "order" attribute values.
520         decorators.sort(Comparator.comparing(DecoratorAndOrder::order));
521 
522         return decorators;
523     }

525     /**
526      * Adds decorators to the specified {@code list}. Decorators which are annotated with {@link Decorator}
527      * and user-defined decorators will be collected.
528      */
529     private static void collectDecorators(List<DecoratorAndOrder> list, Annotation[] annotations) {
530         if (annotations.length == 0) {
531             return;
532         }
533 
534         // Respect the order of decorators which is specified by a user. The first one is first applied
535         // for most of the cases. But if @Decorator and user-defined decorators are specified in a mixed order,
536         // the specified order and the applied order can be different. To overcome this problem, we introduce
537         // "order" attribute to @Decorator annotation to sort decorators. If a user-defined decorator
538         // annotation has "order" attribute, it will be also used for sorting.
539         for (final Annotation annotation : annotations) {
540             if (annotation instanceof Decorator) {
541                 final Decorator d = (Decorator) annotation;
542                 list.add(new DecoratorAndOrder(d, newDecorator(d), d.order()));
543                 continue;
544             }
545 
546             if (annotation instanceof Decorators) {
547                 final Decorator[] decorators = ((Decorators) annotation).value();
548                 for (final Decorator d : decorators) {
549                     list.add(new DecoratorAndOrder(d, newDecorator(d), d.order()));
550                 }
551                 continue;
552             }
553 
554             DecoratorAndOrder udd = userDefinedDecorator(annotation);
555             if (udd != null) {
556                 list.add(udd);
557                 continue;
558             }
559 
560             // If user-defined decorators are repeatable and they are specified more than once.
561             try {
562                 final Annotation[] decorators = (Annotation[]) annotation.annotationType()
563                                                                          .getMethod("value")
564                                                                          .invoke(annotation);
565                 for (final Annotation decorator : decorators) {
566                     udd = userDefinedDecorator(decorator);
567                     if (udd == null) {
568                         break;
569                     }
570                     list.add(udd);
571                 }
572             } catch (Throwable ignore) {
573                 // The annotation may be a container of a decorator or may be not, so we just ignore
574                 // any exception from this clause.
575             }
576         }
577     }

579     /**
580      * Returns a decorator with its order if the specified {@code annotation} is one of the user-defined
581      * decorator annotation.
582      */
583     @Nullable
584     private static DecoratorAndOrder userDefinedDecorator(Annotation annotation) {
585         // User-defined decorator MUST be annotated with @DecoratorFactory annotation.
586         final DecoratorFactory d = annotation.annotationType().getAnnotation(DecoratorFactory.class);
587         if (d == null) {
588             return null;
589         }
590 
591         // In case of user-defined decorator, we need to create a new decorator from its factory.
592         @SuppressWarnings("unchecked")
593         final DecoratorFactoryFunction<Annotation> factory = getInstance(d, DecoratorFactoryFunction.class);
594 
595         // If the annotation has "order" attribute, we can use it when sorting decorators.
596         int order = 0;
597         try {
598             final Object value = annotation.annotationType().getMethod("order").invoke(annotation);
599             if (value instanceof Integer) {
600                 order = (Integer) value;
601             }
602         } catch (Throwable ignore) {
603             // A user-defined decorator may not have an 'order' attribute.
604             // If it does not exist, '0' is used by default.
605         }
606         return new DecoratorAndOrder(annotation, factory.newDecorator(annotation), order);
607     }

647     /**
648      * Returns an immutable list builder of objects which are specified as the {@code value} of annotation
649      * {@code T}.
650      */
651     private static <T extends Annotation, R> Builder<R> annotationValues(
652             Method targetMethod, Class<?> targetClass, Class<T> annotationClass, Class<R> expectedType) {
653 
654         requireNonNull(annotationClass, "annotationClass");
655         requireNonNull(targetMethod, "targetMethod");
656         requireNonNull(targetClass, "targetClass");
657 
658         final Builder<R> builder = new Builder<>();
659         for (final T annotation : targetMethod.getAnnotationsByType(annotationClass)) {
660             builder.add(getInstance(annotation, expectedType));
661         }
662         for (final T annotation : targetClass.getAnnotationsByType(annotationClass)) {
663             builder.add(getInstance(annotation, expectedType));
664         }
665         return builder;
666     }

668     /**
669      * Returns a cached instance of the specified {@link Class}.
670      */
671     @SuppressWarnings("unchecked")
672     static <T> T getInstance(Annotation annotation, Class<T> expectedType) {
673         try {
674             final Class<? extends T> clazz = (Class<? extends T>) invokeValueMethod(annotation);
675             return expectedType.cast(instanceCache.computeIfAbsent(clazz, type -> {
676                 try {
677                     final Constructor<? extends T> constructor = clazz.getDeclaredConstructor();
678                     constructor.setAccessible(true);
679                     return constructor.newInstance();
680                 } catch (Exception e) {
681                     throw new IllegalStateException(
682                             "A class specified in @" + annotation.getClass().getSimpleName() +
683                             " annotation must have an accessible default constructor: " + clazz.getName(), e);
684                 }
685             }));
686         } catch (ClassCastException e) {
687             throw new IllegalArgumentException(
688                     "A class specified in @" + annotation.getClass().getSimpleName() +
689                     " annotation cannot be cast to " + expectedType, e);
690         }
691     }
692 
693     /**
694      * Returns an object which is returned by {@code value()} method of the specified annotation {@code a}.
695      */
696     private static Object invokeValueMethod(Annotation a) {
697         try {
698             return a.getClass().getMethod("value").invoke(a);
699         } catch (Exception e) {
700             throw new IllegalStateException("An annotation @" + a.getClass().getSimpleName() +
701                                             " must have a 'value' method", e);
702         }
703     }

707     /**
708      * A {@link PathMapping} implementation that combines path prefix and another {@link PathMapping}.
709      */
710     @VisibleForTesting
711     static final class PrefixAddingPathMapping extends AbstractPathMapping {
712 
713         private final String pathPrefix;
714         private final PathMapping mapping;
715         private final String loggerName;
716         private final String meterTag;
717 
718         PrefixAddingPathMapping(String pathPrefix, PathMapping mapping) {
719             assert mapping instanceof GlobPathMapping || mapping instanceof RegexPathMapping
720                     : "unexpected mapping type: " + mapping.getClass().getName();
721 
722             this.pathPrefix = pathPrefix;
723             this.mapping = mapping;
724             loggerName = loggerName(pathPrefix) + '.' + mapping.loggerName();
725             meterTag = PrefixPathMapping.PREFIX + pathPrefix + ',' + mapping.meterTag();
726         }
727 
728         @Override
729         protected PathMappingResult doApply(PathMappingContext mappingCtx) {
730             final String path = mappingCtx.path();
731             if (!path.startsWith(pathPrefix)) {
732                 return PathMappingResult.empty();
733             }
734 
735             final PathMappingResult result =
736                     mapping.apply(mappingCtx.overridePath(path.substring(pathPrefix.length() - 1)));
737             if (result.isPresent()) {
738                 return PathMappingResult.of(path, mappingCtx.query(), result.pathParams());
739             } else {
740                 return PathMappingResult.empty();
741             }
742         }
743 
744         @Override
745         public Set<String> paramNames() {
746             return mapping.paramNames();
747         }
748 
749         @Override
750         public String loggerName() {
751             return loggerName;
752         }
753 
754         @Override
755         public String meterTag() {
756             return meterTag;
757         }
758 
759         @Override
760         public boolean equals(Object o) {
761             if (this == o) {
762                 return true;
763             }
764             if (!(o instanceof PrefixAddingPathMapping)) {
765                 return false;
766             }
767 
768             final PrefixAddingPathMapping that = (PrefixAddingPathMapping) o;
769             return pathPrefix.equals(that.pathPrefix) && mapping.equals(that.mapping);
770         }
771 
772         @Override
773         public int hashCode() {
774             return 31 * pathPrefix.hashCode() + mapping.hashCode();
775         }
776 
777         @Override
778         public String toString() {
779             return '[' + PrefixPathMapping.PREFIX + pathPrefix + ", " + mapping + ']';
780         }
781     }

783     /**
784      * A default exception handler is used when a user does not specify exception handlers
785      * by {@link ExceptionHandler} annotation.
786      */
787     private static class DefaultExceptionHandler implements ExceptionHandlerFunction {
788         @Override
789         public HttpResponse handleException(RequestContext ctx, HttpRequest req, Throwable cause) {
790             if (cause instanceof IllegalArgumentException) {
791                 return HttpResponse.of(HttpStatus.BAD_REQUEST);
792             }
793 
794             if (cause instanceof HttpStatusException) {
795                 return HttpResponse.of(((HttpStatusException) cause).httpStatus());
796             }
797 
798             if (cause instanceof HttpResponseException) {
799                 return ((HttpResponseException) cause).httpResponse();
800             }
801 
802             return ExceptionHandlerFunction.DEFAULT.handleException(ctx, req, cause);
803         }
804     }
